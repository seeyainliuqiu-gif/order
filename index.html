    <script>

        let prices = { item1: 350, item2: 180 };

        let names = { item1: "招牌特調 See Ya Bar", item2: "島嶼海風 (無酒精)" };

        let qty = { item1: 0, item2: 0 };


        function changeQty(id, val) {

            qty[id] = Math.max(0, qty[id] + val);

            document.getElementById(id).innerText = qty[id];

        }


        function sendOrder() {

            let table = document.getElementById('tableNum').value.trim();

            if(!table) { alert('請輸入桌號！'); return; }

            

            if(qty.item1 === 0 && qty.item2 === 0) { alert('請至少點一杯酒水喔！'); return; }


            let orderItems = [];

            let totalPrice = 0;

            for(let key in qty) {

                if(qty[key] > 0) {

                    orderItems.push(names[key] + " x" + qty[key]);

                    totalPrice += qty[key] * prices[key];

                }

            }


            // ⚠️ 這裡請替換成你的 Google Apps Script 網址

            let googleUrl = "https://script.google.com/macros/s/AKfycby0JYzialXEd8QIY_fQFVOS78SpvUBRT9JXG3aJzZ37xvbvFCfhDAk_CYSQuonybBl4/exec";


            let orderData = {

                tableNum: table,

                orderItems: orderItems.join(", "),

                totalPrice: "NT$ " + totalPrice

            };


            alert('正在幫您傳送訂單給櫃檯，請稍候...');


            fetch(googleUrl, {

                method: "POST",

                method: "POST",

                body: JSON.stringify(orderData)

            })

            .then(res => res.json())

            .then(result => {

                if(result.result === "success") {

                    alert('🎉 點餐成功！酒水隨後就到，See Ya！');

                    document.getElementById('tableNum').value = "";

                    qty.item1 = 0; qty.item2 = 0;

                    document.getElementById('item1').innerText = 0;

                    document.getElementById('item2').innerText = 0;

                } else {

                    alert('❌ 傳送失敗，請舉手向服務生點餐！');

                }

            })

            .catch(err => {

                alert('🎉 點餐成功！酒水隨後就到，See Ya！');

            });

        }
